\usepackage{tikz}
	\usetikzlibrary{arrows}

\newcommand\LongitudePlane[4][current plane]{%{#2} Elevation,  {#3} Azimuth, {#4} Longitude.
	\tikzset{ #1/.style = {cm = { { sin(#3)*cos(#4) + cos(#3)*sin(#4)}, {-cos(#3)*sin(#2)*cos(#4) + sin(#3)*sin(#2)*sin(#4)}, 0, cos(#2), (0,0)} }}
}

\newcommand\LatitudePlane[5][current plane]{%{#2} Radius, {#3} Elevation,  {#4} Azimuth, {#5} Latitude.
	\tikzset{ #1/.style = {cm = { sin(#4), -cos(#4)*sin(#3), cos(#4), sin(#4)*sin(#3), (0, {(#2)*cos(#3)*sin(#5)}) } }}
}

\newcommand\XLatitudePlane[5][current plane]{%{#2} Radius, {#3} Elevation,  {#4} Azimuth, {#5} Latitude.
	\tikzset{ #1/.style = {cm = { cos(#4), {sin(#4)*sin(#3)}, 0, cos(#3), ({(#2)*sin(#4)*sin(#5)}, {-(#2)*cos(#4)*sin(#3)*sin(#5)}) } }}
}

\newcommand\YLatitudePlane[5][current plane]{%{#2} Radius, {#3} Elevation,  {#4} Azimuth, {#5} Latitude.
	\tikzset{ #1/.style = {cm= { sin(#4), {-cos(#4)*sin(#3)}, 0, cos(#3), ({(#2)*cos(#4)*sin(#5)}, {(#2)*sin(#4)*sin(#3)*sin(#5)}) } }}
}

\newcommand\defcommonplanes[2]{% {#1} Elevation, {#2} Azimuth
\tikzset{xyplane/.style={cm = { sin(#2), {-cos(#2)*sin(#1)}, cos(#2), {sin(#2)*sin(#1)}, (0, 0) } }}
\tikzset{yzplane/.style={cm = { cos(#2), { sin(#2)*sin(#1)}, 0, cos(#2), (0,0) } }}
\tikzset{xzplane/.style={cm = { sin(#2), {-cos(#2)*sin(#1)}, 0, cos(#2), (0,0) } }}
}

\newcommand\DrawLongitudeCircle[4][1]{
  \LongitudePlane{#2}{#3}{#4}
   % angle of "visibility"
  \pgfmathsetmacro\angVis{atan(sin(#4)*cos(#2)/sin(#2))}
  \draw[current plane] (\angVis: {#1}) arc (\angVis:\angVis+180: {#1});
  \draw[current plane,dashed] (\angVis-180: {#1}) arc (\angVis-180:\angVis: {#1});
}

\newcommand\DrawXLatitudeCircle[4][1]{
  \XLatitudePlane{#1}{#2}{#3}{#4}
  \pgfmathsetmacro\sinVis{sin(#4)/cos(#4)*sin(#2)/cos(#2)}
  % angle of "visibility"
  \pgfmathsetmacro\angVis{asin(min(1,max(\sinVis,-1)))}
  \draw[current plane] (\angVis:{(#1)*cos(#4)}) arc (\angVis:-\angVis-180:{(#1)*cos(#4)});
  \draw[current plane,dashed] (180-\angVis:{(#1)*cos(#4)}) arc (180-\angVis:\angVis:{(#1)*cos(#4)});
}

\newcommand\DrawYLatitudeCircle[4][1]{
  \YLatitudePlane{#1}{#2}{#3}{#4}
  \pgfmathsetmacro\sinVis{sin(#4)/cos(#4)*sin(#2)/cos(#2)}
  % angle of "visibility"
  \pgfmathsetmacro\angVis{asin(min(1,max(\sinVis,-1)))}
  \draw[current plane] (\angVis:{(#1)*cos(#4)}) arc (\angVis:-\angVis-180:{(#1)*cos(#4)});
  \draw[current plane,dashed] (180-\angVis:{(#1)*cos(#4)}) arc (180-\angVis:\angVis:{(#1)*cos(#4)});
}
\newcommand\DrawLatitudeCircle[4][1]{
	\LatitudePlane{#1}{#2}{0}{#4} % To make calulations easiear assume \angAz=0.
    \begin{scope}[thin]
	\clip[current plane] ({(#1)*tan(#2)*sin(#4)}, {-2*(#1)}) rectangle ({2*(#1)}, {2*(#1)});
	\draw[current plane] (0,0) circle ({(#1)*cos(#4)});
	\end{scope}
	\begin{scope}[thin]
	\draw[current plane] ({-2*(#1)},{-2*(#1)}) rectangle ({(#1)*tan(#2)*sin(#4)},{2*(#1)});
	\draw[current plane, dashed] (0,0) circle ({(#1)*cos(#4)});
	\end{scope}
}
